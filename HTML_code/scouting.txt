	function auto_grow(element) {

<script type="text/javascript">

	function consulta(elemento, simbolo, forma, textoarea){
			var valor = document.getElementById(elemento).value;     
			$.ajax({
				url:'../PHP/scouting_example.php',
				method:'POST',
				data:{
					valor:valor ,
					variable : elemento,
					simbolo: simbolo	
				}
			})
			.done(function(eco){
			  var x = document.getElementById(forma);
			  if (x.style.display === "none") {x.style.display = "block";} else {x.style.display = "none";}
			  document.getElementById(textoarea).value = eco;
			  var element = document.getElementById(textoarea);
			  auto_grow(element); 
			});
	}
</script>
